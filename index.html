import streamlit as st from datetime import date from streamlit_gsheets import GSheetsConnection import pandas as pd # ========================================== # 1. 환경 설정 및 상수 (Config) # ========================================== # 💡 앱 이름과 버전을 분리했습니다. (로직: V4.0 / 앱 버전: v1.5) st.set_page_config(page_title="TQQQ 무한매수 V4.0 (계산기 v1.5)", page_icon="📈", layout="wide") SHEET_URL = "https://docs.google.com/spreadsheets/d/1eLf1envTkRq9fGx2H9c_btI2Z18OqL1L71WBhXHjOTY/edit?gid=0#gid=0" # ========================================== # 2. 데이터 처리 모듈 (Data Layer) # ========================================== def get_conn(): return st.connection("gsheets", type=GSheetsConnection) def fetch_latest_status(sheet_name): try: df = get_conn().read(spreadsheet=SHEET_URL, worksheet=sheet_name, ttl=0) return df.iloc[-1] if not df.empty else None except: return None def update_status(sheet_name, t_val, balance, avg_price): new_data = pd.DataFrame([{"T값": t_val, "잔고": balance, "평단가": avg_price}]) get_conn().update(spreadsheet=SHEET_URL, worksheet=sheet_name, data=new_data) st.cache_data.clear() def fetch_records(sheet_name): return get_conn().read(spreadsheet=SHEET_URL, worksheet=sheet_name, ttl=0) def append_record(sheet_name, end_date, total_invested, final_profit): profit_rate = f"{round((final_profit/total_invested)*100, 2)}%" if total_invested > 0 else "0%" new_entry = pd.DataFrame([{ "종료일": str(end_date), "투입원금": total_invested, "수익금": final_profit, "수익률": profit_rate }]) conn = get_conn() existing_df = fetch_records(sheet_name) updated_df = pd.concat([existing_df, new_entry], ignore_index=True) conn.update(spreadsheet=SHEET_URL, worksheet=sheet_name, data=updated_df) st.cache_data.clear() # ========================================== # 3. 비즈니스 로직 모듈 (Logic Layer) # ========================================== def calculate_orders(current_t, current_bal, current_avg): if current_t == 0 or current_avg == 0: return 0, 0.0, 0.0 star_pct = (20 - current_t) * 0.01 star_point = current_avg * (1 + star_pct) buy_qty = round((current_bal / (40 - current_t) / 2) / current_avg) if current_t < 40 else 0 sell_price = current_avg * 1.10 if current_t < 20 else star_point return buy_qty, star_point, sell_price # ========================================== # 4. UI 렌더링 모듈 (View Layer) # ========================================== def apply_theme(account_choice): if account_choice == "진": st.markdown(""" """, unsafe_allow_html=True) # ========================================== # 5. 메인 앱 실행 (App Entry Point) # ========================================== def main(): st.sidebar.title("👤 계좌 설정") account_choice = st.sidebar.radio("운용할 계좌를 선택하세요:", ["웅", "진"]) apply_theme(account_choice) suffix = "1" if account_choice == "웅" else "2" status_sheet = f"내상태{suffix}" record_sheet = f"기록{suffix}" last_status = fetch_latest_status(status_sheet) if last_status is not None: default_t, default_bal, default_avg = float(last_status['T값']), float(last_status['잔고']), float(last_status['평단가']) else: default_t, default_bal, default_avg = 0.0, (10000.0 if account_choice == "진" else 4000.0), 0.0 # 💡 메인 타이틀도 명확하게 분리했습니다. st.title(f"📈 TQQQ 무한매수 V4.0 (App v1.5)") st.write(f"현재 **{account_choice}** 계좌 데이터를 관리하고 있습니다.") tab1, tab2 = st.tabs(["🛒 오늘의 주문 계산", "📝 사이클 종료 정산 (차트)"]) with tab1: col1, col2 = st.columns([1.5, 1]) with col1: st.subheader("☀️ 1단계: 어제장 체결 내용 업데이트 (아침)") st.caption(f"💾 현재 시트 데이터: T값 **{default_t}** / 잔금 **\${default_bal:,.2f}** / 평단 **\${default_avg:,.2f}**") with st.container(border=True): c1, c2, c3 = st.columns(3) add_t = c1.selectbox("어제 매수 체결 (T값 증가)", [0.0, 0.5, 1.0], index=0) spent_dollar = c2.number_input("어제 체결 금액 차감 ($)", min_value=0.0, step=10.0, value=0.0) new_avg = c3.number_input("오늘 내 평단가 ($)", value=default_avg, step=0.01) current_t = default_t + add_t current_bal = default_bal - spent_dollar current_avg = new_avg st.write("") if st.button(f"💾 1단계 완료: 체결 내용 시트에 저장하기", type="primary", use_container_width=True): try: update_status(status_sheet, current_t, current_bal, current_avg) st.success("✅ 구글 시트가 최신 데이터로 업데이트되었습니다!") st.rerun() except Exception as e: st.error(f"저장 실패: {e}") st.divider() st.subheader("🌙 2단계: 오늘의 예약 주문 세팅 (밤)") st.write(f"👉 **주문 기준 상태:** T값 **{default_t}** / 잔금 **\${default_bal:,.2f}**") if default_t == 0: st.info("✨ 새로운 사이클의 시작입니다! 오늘은 앱 계산 없이 **하루 시드만큼 시장가 매수**를 진행하세요.") st.write(f"👉 추천 매수 금액: **\${default_bal / 40:,.2f}** (총 예산의 1/40)") else: buy_qty, star_point, sell_price = calculate_orders(default_t, default_bal, default_avg) if buy_qty > 0: st.info(f"**🟢 [매수 1] 평단 LOC:** {round(default_avg, 2)} / {buy_qty}주") st.info(f"**🟢 [매수 2] 별지점 LOC:** {round(star_point, 2)} / {buy_qty}주") else: st.warning("T값이 40에 도달하여 추가 매수하지 않습니다. (매도만 진행)") st.error(f"**🔴 [매도] 전량 익절/탈출 LOC:** {round(sell_price, 2)}") st.write("") with st.expander("✨ 새 사이클 시작 또는 강제 숫자 수정"): st.subheader("🆕 새로운 사이클 시작") new_seed = st.number_input("시작 잔고 입력 ($)", value=10000.0 if account_choice=="진" else 4000.0, step=100.0) if st.button(f"🚀 {account_choice} 계좌 초기화"): try: update_status(status_sheet, 0.0, new_seed, 0.0) st.success("✅ 계좌가 초기화되었습니다. 첫 매수를 진행하세요!") st.rerun() except Exception as e: st.error(f"초기화 실패: {e}") st.divider() st.subheader("⚙️ 수동 숫자 수정 (계산 꼬였을 때)") c_man1, c_man2, c_man3 = st.columns(3) m_t = c_man1.number_input("T값 수정", value=default_t, step=0.5) m_b = c_man2.number_input("잔금 수정 ($)", value=default_bal, step=10.0) m_a = c_man3.number_input("평단가 수정 ($)", value=default_avg, step=0.01) if st.button("수동 값으로 덮어쓰기"): update_status(status_sheet, m_t, m_b, m_a) st.rerun() with col2: st.info(f"**💡 아주 심플한 매매 루틴**\n\n**[아침]**\n토스 알림을 보고 간밤에 체결된 내역을 1단계에 적은 뒤 **[저장]**을 누릅니다.\n\n**[밤]**\n토스증권을 켜고 2단계 구역에 떠 있는 수량과 가격대로 예약 주문을 걸고 취침합니다.") with tab2: st.subheader(f"📊 {account_choice} 누적 수익금 추이") try: record_df = fetch_records(record_sheet) if not record_df.empty and "수익금" in record_df.columns: chart_df = record_df.dropna(subset=["수익금"]).copy() chart_df["수익금"] = pd.to_numeric(chart_df["수익금"], errors="coerce").fillna(0) chart_df["누적수익금"] = chart_df["수익금"].cumsum() if "종료일" in chart_df.columns: chart_df["종료일"] = chart_df["종료일"].astype(str) chart_df.set_index("종료일", inplace=True) st.success(f"🎉 **현재까지 총 누적 수익금: \${chart_df['수익금'].sum():,.2f}**") st.area_chart(chart_df["누적수익금"]) except: st.info("기록을 불러오는 중입니다.") st.divider() st.subheader(f"💾 {account_choice} 사이클 종료 기록") end_date = st.date_input("종료일", date.today()) total_invested = st.number_input("총 투입 원금 ($)", min_value=0.0, value=1000.0) final_profit = st.number_input("실현 수익금 ($)", value=100.0) if st.button(f"💾 {account_choice} 최종 결과 저장"): try: append_record(record_sheet, end_date, total_invested, final_profit) st.success("기록 완료!") st.balloons() except Exception as e: st.error(f"오류: {e}") if __name__ == "__main__": main()
